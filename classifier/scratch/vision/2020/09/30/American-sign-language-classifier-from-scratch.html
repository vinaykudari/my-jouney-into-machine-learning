def cross_entropy_loss(predictions, labels): 
    activations = log_softmax(predictions)
    loss = activations.gather(1, labels.unsqueeze(-1)).mean()
    return loss

Optimise weights 

def train_epoch(dl, weights, bias, learning_rate):
    for x, y in dl:
        preds = get_predictions(x, weights, bias)
        loss = cross_entropy_loss(preds, y)
        loss.backward()
        weights.data = weights.data - weights.grad*learning_rate
        bias.data -= bias.grad*learning_rate
        weights.grad.zero_()
        bias.grad.zero_()
    print(f'loss={loss}')
    return weights, bias
