def download_and_extract():
    dest_directory = DATA_DIR
    
    if not os.path.exists(dest_directory):
        os.makedirs(dest_directory)
    filename = DATA_URL.split('/')[-1]
    filepath = os.path.join(dest_directory, filename)
    
    if not os.path.exists(filepath):
        def _progress(count, block_size, total_size):
            download_per = float(count * block_size) / float(total_size) * 100.0
            if round(download_per, 2) == round(int(download_per), 2):
                sys.stdout.write(f'\rDownloading {filename} {int(download_per)}%')
                sys.stdout.flush()
        filepath, _ = urllib.request.urlretrieve(DATA_URL, filepath, reporthook=_progress)
    
    folder_name = filename.split(".")[0]
    if not os.path.exists(DATA_DIR/folder_name):
        sys.stdout.write(f'Extracting {filename} into {folder_name}')
        tarfile.open(filepath, 'r:gz').extractall(dest_directory)
        sys.stdout.flush()
        
    print('STL-10 dataset is ready')
    return DATA_DIR/folder_name
        
DATASET_PATH = download_and_extract()

class STL10Image(fastuple):
    def show(self, ctx=None, **kwargs):
        image, title = self
        if not isinstance(image, Tensor):
            image = tensor(image)
        return show_image(image.permute(2,1,0), title=title, ctx=ctx, **kwargs)

class STL10Transform(Transform):
    def __init__(
        self, path, image_fname='train_X.bin', channel=3, width=96, height=96,
        labels_fname='train_y.bin', classes_fname='class_names.txt', *args, **kwargs,
    ):
        self.image_file = open(path/image_fname, 'rb')
        self.image_array = np.fromfile(self.image_file, dtype=np.uint8)
        self.images = torch.reshape(tensor(self.image_array).float()/255., (-1, channel, width, height))
        
        self.label_file = open(path/labels_fname)
        self.labels = np.fromfile(self.label_file, dtype=np.uint8)
        
        self.classes_file = open(path/classes_fname)
        self.classes = self.classes_file.read().splitlines()
        
    def encodes(self, idx):
        img = self.images[idx]
        label = self.labels[idx]
        return STL10Image(img, self.classes[label-1])

    def __init__(self, encoder, head):
        self.encoder, self.head = encoder, head

        ftrs = self.encoder(x1)

        channel=3, width=96, height=96, split=80, *args, **kwargs,
    ):
        self.image_file = open(path/image_fname, 'rb')
        self.image_array = np.fromfile(self.image_file, dtype=np.uint8)
        self.images = torch.reshape(
            tensor(self.image_array).float()/255.,
            (-1, channel, width, height),
        )
        self.dtype = dtype
        self.total_samples = len(self.images)
        self.train_samples = (self.total_samples * split) // 100
        self.valid_samples = self.total_samples - self.train_samples

        self.labels = np.fromfile(self.label_file, dtype=np.uint8)

        self.classes = self.classes_file.read().splitlines()
    
    def __len__(self): 
        if self.dtype == 'train':
            return self.train_samples
        return self.valid_samples
    
    def __getitem__(self, i):
        if self.dtype != 'train' or self.dtype == 'valid':
            # Picking a valid example
            if i >= self.valid_samples:
                raise IndexError("list index out of range")
            i += self.train_samples
        else:
            # Picking a training example
            if i >= self.train_samples:
                raise IndexError("list index out of range")

class DataLoader:
    def __init__(self, ds, bs=128, shuffle=False, n_workers=1):
        self.ds,self.bs,self.shuffle,self.n_workers = ds,bs,shuffle,n_workers

    def __len__(self): 
        return (len(self.ds)-1)//self.bs+1

    def __iter__(self):
        idxs = L.range(self.ds)
        if self.shuffle: idxs = idxs.shuffle()
        chunks = [idxs[n:n+self.bs] for n in range(0, len(self.ds), self.bs)]
